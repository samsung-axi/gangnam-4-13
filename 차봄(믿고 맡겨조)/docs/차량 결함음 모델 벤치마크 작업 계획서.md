```python
    "scheduler": "CosineAnnealing",

    "early_stop_patience": 5, # 5 에폭 연속 개선 없으면 중단
}

# ──────────── Seed 완전 고정 (모든 파일에 적용) ────────────
import torch, numpy as np, random

def set_seed(seed=42):
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    torch.cuda.manual_seed_all(seed)
    torch.backends.cudnn.deterministic = True   # 재현성 보장
    torch.backends.cudnn.benchmark = False       # 속도 약간 희생 → 재현성 확보

set_seed(42)  # 모든 train_*.py 시작 시 호출
```

> [!IMPORTANT]
> **Baseline은 2 에폭이 아니라 10 에폭** 돌려야 합니다. Head가 수렴하는 데 5-10 에폭이 필요하고, 2 에폭이면 아직 학습이 덜 된 상태라 baseline 의미가 없습니다.

### Scheduler 전략 (⚠️ CosineAnnealing + EarlyStopping 충돌 방지)

```python
# Baseline: ReduceLROnPlateau (EarlyStopping과 호환)
#   → macro_f1이 정체되면 lr을 줄이므로 EarlyStopping과 자연스럽게 협력
scheduler_baseline = torch.optim.lr_scheduler.ReduceLROnPlateau(
    optimizer, mode='max', factor=0.5, patience=2, verbose=True
)

# Fine-tune: CosineAnnealing (전체 에폭까지 돌리는 것이 전제)
scheduler_finetune = torch.optim.lr_scheduler.CosineAnnealingLR(
    optimizer, T_max=finetune_epochs
)
```

> [!WARNING]
> CosineAnnealing은 epoch 끝까지 가는 걸 전제합니다. EarlyStopping으로 중간에 끊으면 cosine이 반도 못 돌 수 있으므로 **Baseline에서는 ReduceLROnPlateau를, Fine-tune에서만 CosineAnnealing을 사용**합니다.

## Q3. Loss 가중치

현재 데이터 분포에서 계산:

```python
# 데이터 수 (Train)
counts = {"normal": 400, "engine": 334, "brake": 132, "starter": 90}
total = 956

# Inverse Frequency 가중치 (자동 계산)
weights = {
    "normal":  total / (4 * 400),  # = 0.60
    "engine":  total / (4 * 334),  # = 0.72
    "brake":   total / (4 * 132),  # = 1.81
    "starter": total / (4 *  90),  # = 2.66
}

# PyTorch 적용 — 모든 모델에 동일하게!
class_weights = torch.tensor([0.60, 0.72, 1.81, 2.66])
criterion = nn.CrossEntropyLoss(weight=class_weights.to(device))
```

> [!TIP]
> 이 공식 `total / (num_classes × count)`는 sklearn의 `compute_class_weight('balanced')`와 동일합니다. **모든 모델에 동일하게 적용**해야 공정한 비교가 됩니다.

## 속도 측정 방법 (정석)

```python
import time
import torch

def measure_latency(model, sample_input, device, n_warmup=10, n_runs=100):
    """
    정확한 추론 속도 측정
    - batch_size=1 (실제 배포 환경과 동일)
    - warm-up 10회 후 100회 평균
    """
    model.eval()
    sample_input = sample_input.to(device)
    
    # ⚠️ AMP 끄기 — fp32로 측정해야 ONNX/실배포 latency와 일치
    # Warm-up (GPU 초기화 오버헤드 제거)
    with torch.no_grad(), torch.cuda.amp.autocast(enabled=False):
        for _ in range(n_warmup):
            model(sample_input)
    
    # 동기화 후 측정 (GPU 비동기 실행 보정)
    torch.cuda.synchronize()
    times = []
    with torch.no_grad(), torch.cuda.amp.autocast(enabled=False):
        for _ in range(n_runs):
            torch.cuda.synchronize()
            start = time.perf_counter()
            model(sample_input)
            torch.cuda.synchronize()
            elapsed = (time.perf_counter() - start) * 1000  # ms
            times.append(elapsed)
    
    avg = sum(times) / len(times)
    print(f"Latency: {avg:.1f}ms (avg of {n_runs} runs, GPU: {torch.cuda.get_device_name()})")
    return avg
```

> [!IMPORTANT]
> `torch.cuda.synchronize()`가 핵심입니다. GPU는 비동기 실행이라 이 호출 없이 시간을 재면 실제보다 빠르게 나옵니다.

```python
    id2label={i: l for i, l in enumerate(LABEL_LIST)},
    ignore_mismatched_sizes=True  # ← 527→4 head 교체
).to(DEVICE)
```

### 입력 형식

```python
import torchaudio
import torch.nn.functional as F

# ⚠️ waveform을 미리 5초로 padding/truncate (벤치마크 입력 길이 통일)
target_len = 16000 * 5  # 5초
if waveform.shape[-1] < target_len:
    waveform = F.pad(waveform, (0, target_len - waveform.shape[-1]))
else:
    waveform = waveform[..., :target_len]

mel_transform = torchaudio.transforms.MelSpectrogram(
    sample_rate=16000, n_fft=1024,
    win_length=400, hop_length=160, n_mels=128
)

# waveform: (B, T)
mel = mel_transform(waveform)           # (B, 128, time)
mel = torch.log(mel + 1e-6)
mel = mel.unsqueeze(1)                  # (B, 1, 128, time) ← PaSST 입력
mel = mel.to(DEVICE)
```

### 전체 모델을 하나로 감싸기

```python
class PaSSTClassifier(nn.Module):
    def __init__(self, arch="passt_n_s", num_classes=4):
        super().__init__()
        self.arch = arch
        self.backbone = get_basic_model(arch=arch, pretrained=True)
        # ⚠️ 하드코딩(768) 대신 동적으로 embed_dim 가져오기
        feat_dim = getattr(self.backbone, 'embed_dim', 768)
        self.classifier = nn.Linear(feat_dim, num_classes)

    def forward(self, mel):
        # mel: (B, 1, 128, time)
        features = self.backbone(mel)   # (B, 768) or (B, T, 768)
        # ⚠️ hear21passt 버전에 따라 (B, T, 768)이 나올 수 있음 → pool 필요
        if features.dim() == 3:
            features = features.mean(dim=1)  # (B, T, 768) → (B, 768)
        return self.classifier(features)  # (B, num_classes)

# PaSST-N-S 변종
model_ns = PaSSTClassifier(arch="passt_n_s", num_classes=NUM_CLASSES).to(DEVICE)

# PaSST-S-SWA 변종
model_swa = PaSSTClassifier(arch="passt_s_swa_p16_128_ap476", num_classes=NUM_CLASSES).to(DEVICE)
```

### Baseline vs Fine-tune

```python
# ─── Baseline: backbone 고정, classifier만 학습 ───
for p in model.backbone.parameters():
    p.requires_grad = False           # backbone 🔒
for p in model.classifier.parameters():
    p.requires_grad = True

# ⚠️ LayerNorm + pos_embed 열어두기 (domain shift 적응 가속)
# Baseline 성능이 너무 낮으면 이게 원인일 확률 높음
for name, p in model.backbone.named_parameters():
    if "norm" in name.lower() or "pos_embed" in name.lower():
        p.requires_grad = True

optimizer = torch.optim.AdamW(
    filter(lambda p: p.requires_grad, model.parameters()), lr=1e-3
)
# epochs = 10

# ─── Fine-tune: 전체 학습 ───
for p in model.parameters():
    p.requires_grad = True            # 전체 🔓

optimizer = torch.optim.AdamW(model.parameters(), lr=3e-5)
# epochs = 20
```
